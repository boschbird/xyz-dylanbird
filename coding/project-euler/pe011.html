What is the greatest product of four adjacent numbers in the same direction (up, down, left, right, or diagonally) in the 20Ã—20 grid? Run this solution at repl.io here. 

      
        import numpy as np

        #0. turn textfile into a matrix
        #1. find max product of adjacent horizontal numbers
        #2. find max product of adjacent vertical numbers
        #3. find max product of adjacent (forward) diagonal numbers
        #4. find max product of adjacent (backward) diagonal numbers
        #5. compare products 1. through to 4. to fine the largest

        with open("data.txt") as data:
          txt = data.read()
          grid = [line for line in txt.splitlines()]
          grid = [line.split() for line in grid]
          grid = [[int(i) for i in j] for j in grid]


        A = np.array(grid)

        horizprod = 1
        maxhoriz = 1
        for i in range(0,20):
          for j in range(0,20-4):
            for n in A[i,j:j+4]:
              horizprod *= n
            if horizprod > maxhoriz:
              maxhoriz = horizprod
            horizprod = 1

        vertprod = 1
        maxvert = 1
        for j in range(0,20):
          for i in range(0,20-4):
            for n in A[i:i+4,j]:
              vertprod *= n
            if vertprod > maxvert:
              maxvert = vertprod
            vertprod = 1

        ldiagprod = 1
        maxldiag = 1
        for i in range(0,20-4):
          for j in range (0,20-4):
            ldiagprod = A[i][j]*A[i+1][j+1]*A[i+2][j+2]*A[i+3][j+3]
            if ldiagprod > maxldiag:
              maxldiag = ldiagprod

        rdiagprod = 1
        maxrdiag = 1
        for i in range(3,20):
          for j in range(0,20-4):
            rdiagprod = A[i][j]*A[i-1][j+1]*A[i-2][j+2]*A[i-3][j+3]
            if rdiagprod > maxrdiag:
              maxrdiag = rdiagprod

        print(max(maxhoriz,maxvert,maxldiag,maxrdiag))
